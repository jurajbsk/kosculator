File -> TopStmnt+
TopStmnt -> StructDef | FuncDef

FuncDef -> Type Ident LParen Args RParen FuncBody
Args -> Null | Type (Comma Type)* Comma?
FuncBody -> LBrack FuncStmnt* RetStmnt FuncStmnt* RBrack
FuncStmnt -> StmntType LBrack FuncStmnt* RBrack | FuncStmntExpr
FuncExprStmnt -> ExprStmnt | RetStmnt

RetStmnt -> RetKeyword Expr? SemiCol

Stmnt -> StmntType LBrack Stmnt* RBrack | ExprStmnt SemiCol
StmntType -> IfStmnt | ForStmnt | ForeachStmnt | ... 
ExprStmnt -> TopStmnt | ...

Expr -> Var | Literal | ...
